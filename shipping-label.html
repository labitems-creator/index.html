  <script>


    // ===== BASIC COMMANDS =====


    function execCmd(command) {


      document.execCommand(

        command,

        false,

        null

      );

    }


    // ===== COLOR =====


    function changeColor(color) {


      document.execCommand(

        "styleWithCSS",

        false,

        true

      );


      document.execCommand(

        "foreColor",

        false,

        color

      );

    }


    // ===== FONT FAMILY =====


    function changeFont(font) {


      applyStyle({

        fontFamily: font

      });

    }


    // ===== FONT SIZE =====


    function changeFontSize(size) {


      applyStyle({

        fontSize: size,

        lineHeight: "1.25"

      });

    }


    // ===== APPLY STYLE =====


    function applyStyle(styles) {


      const selection =

        window.getSelection();


      if (!selection.rangeCount) return;


      const range =

        selection.getRangeAt(0);


      if (range.collapsed) return;


      const span =

        document.createElement("span");


      Object.assign(

        span.style,

        styles

      );


      try {


        const content =

          range.extractContents();


        span.appendChild(content);


        range.insertNode(span);


        selection.removeAllRanges();


      }


      catch (err) {


        alert(

          "Please select a simpler text portion."

        );

      }

    }


    // ===== CLEAR ALL FORMATTING =====


    function clearFormatting() {


      const editor =

        document.getElementById(

          "editableArea"

        );


      const scaleContainer =

        document.getElementById(

          "scaleContainer"

        );


      // Preserve line breaks


      const lines =

        editor.innerText.split("\n");


      // Rebuild clean HTML


      editor.innerHTML =

        lines.join("<br>");


      // Reset scale


      scaleContainer.style.transform =

        "scale(1)";

    }


    // ===== AUTO FIT =====


    function autoFitContent() {


      const wrapper =

        document.querySelector(

          ".wrapper"

        );


      const scaleContainer =

        document.getElementById(

          "scaleContainer"

        );


      // Reset scale first


      scaleContainer.style.transform =

        "scale(1)";


      let scale = 1;


      // Overflow checker


      function isOverflowing() {


        return (

          scaleContainer.getBoundingClientRect().height >

          wrapper.clientHeight - 10

        );

      }


      // Gradually shrink


      while (

        isOverflowing() &&

        scale > 0.4

      ) {


        scale -= 0.01;


        scaleContainer.style.transform =

          `scale(${scale})`;

      }

    }
